<script>
import * as echarts from 'echarts';

export default {
  name: 'EChartsComponent',
  mounted() {
    this.initChart();
  },
  methods: {
    initChart() {
      const chartDom = document.getElementById('main1');
      const myChart = echarts.init(chartDom);
      const option = {
        title: {
          text: 'PID DIstribution',
          left: '4%',
          top: '4%'
        },
        legend: {
          data: ['YBK(Seed) Qty'],
          bottom: '1%'
        },
        grid: {
          left: '6%',
          right: '4%',
          bottom: '8%',
          containLabel: true
        },
        xAxis: {
          type: 'category',
          data: ['PID2', 'PID1', 'PID3', 'PID4', 'PID5', 'PID6', 'PID7'],
          axisTick: {
            show: false
          },
          axisLine: {
            lineStyle: {
              color: 'rgba(0, 0, 0, 0.3)'
            }
          }
        },
        yAxis: {
          type: 'value',
          min: 0,
          max: 100,
          interval: 20,
          axisLabel: {
            color: 'rgba(0, 0, 0, 0.4)' 
          }
        },
        series: [
          {
            name: 'YBK(Seed) Qty',
            data: [60, 30, 20, 20, 20, 20, 20],
            type: 'bar',
            barWidth: '20%' 
          }
        ]
      };
      option && myChart.setOption(option);
    }
  }
};
</script>
