<template>
  <v-layout row wrap>
    <v-flex sm12>
      <v-widget title="SN Find">
        <div slot="widget-content">
          <v-container fluid style="padding-top: 10px;">
            <v-layout row>
              <!-- 标签 -->
              <v-tabs v-model="checktab" grow class="setTabs" active-class="active_tabs">
                <v-tabs-slider color="primary"></v-tabs-slider>
                <v-tab class="NewButton" style="color: #000;">Test Record</v-tab>
                <v-tab class="NewButton" style="color: #000;">Parent/Children</v-tab>
                <!-- sfcSupplement为true才有该功能，即第一、四、十套才有 -->
                <v-tab v-if="sfcSupplement" class="NewButton" style="color: #000;">SYNC to SFC</v-tab>
              </v-tabs>
              <v-spacer></v-spacer>
            </v-layout>
            <!-- 表格 -->
            <v-layout row wrap>
              <v-flex sm12 v-show="checktab === 0" style="margin-top: 24px;">
                <testdata-table :start-date="startDate" :end-date="endDate" :current-time="currenTtime" />
              </v-flex>
              <v-flex sm12 v-if="checktab === 1" style="margin-top: 24px;">
                <parentdata-table :start-date="startDate" :end-date="endDate" :current-time="currenTtime" />
              </v-flex>
              <v-flex sm12 v-if="checktab === 2" style="margin-top: 24px;">
                <sfc-transmission-info :start-date="startDate" :end-date="endDate" :current-time="currenTtime" :now-date-time="nowDateTime" />
              </v-flex>
            </v-layout>
          </v-container>
        </div>
      </v-widget>
    </v-flex>
  </v-layout>
</template>

<script>
import VWidget from '@/components/VWidget';
import moment from 'moment';
import TestdataTable from '../components/snFind/TestdataTable.vue';
import ParentdataTable from '../components/snFind/ParentdataTable.vue';
import SfcTransmissionInfo from '../components/snFind/SFCTransmissionTable.vue';
import { getServicePermission } from '../api/getSFCInformation.js';
import { getCurrentTime } from '../../genius/api/getCurrentTime.js';
import store from '../store';

export default {
  components: {
    VWidget,
    TestdataTable,
    ParentdataTable,
    SfcTransmissionInfo,
    store
  },
  data() {
    return {
      checktab: 0,
      isadmin: false,
      sfcSupplement: false,
      currenTtime: [],
      startDate: '',
      endDate: '',
      nowDateTime: ''
    };
  },
  watch: {
    checktab(newtab) {
      sessionStorage.setItem('route', JSON.stringify({ ...JSON.parse(sessionStorage.getItem('route')), checktab: newtab }));
    },
  },
  created () {
    this.handleChecktab();
    this.getServicePermissionInfor();
    this.getCurrentTimes();
  }, 
  methods: {
    // 刷新维持页签
    handleChecktab() {
      this.isadmin = Buffer.from(this.$cookies.get('admin'), 'base64').toString('utf-8') === 'true' ? true : false;
      let path = JSON.parse(sessionStorage.getItem('route'));
      if (path.from === '/') this.checktab = path.checktab || 0;
    },
    // 获取服务器信息
    async getServicePermissionInfor() {
      const { data: res } = await getServicePermission();
      if (res.status) this.sfcSupplement = res.payload.data;
    }, 
    async getCurrentTimes() {
      const { data: res } = await getCurrentTime();
      if (res.status) {
        this.currenTtime = res.payload.current_time.split('T');
        this.endDate = this.currenTtime[0];
        this.startDate = moment(this.endDate).subtract(6, 'month').format('YYYY-MM-DD');
        this.nowDateTime = moment(res.payload.current_time).subtract(8, 'hour').format('YYYY-MM-DD HH:00');
      }
    },
  }
};
</script>
